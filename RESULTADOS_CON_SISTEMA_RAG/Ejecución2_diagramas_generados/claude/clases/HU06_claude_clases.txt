@startuml
class User {
    - userId: String
    - name: String
    + selectQuestions(): Question[]
    + selectTopics(): Topic[]
    + generatePDF(): PDFDocument
}

class Question {
    - questionId: String
    - content: String
    - category: String
    + getContent(): String
}

class Topic {
    - topicId: String
    - title: String
    - description: String
    + getTitle(): String
}

class PDFDocument {
    - documentId: String
    - title: String
    - content: String
    - creationDate: Date
    - filePath: String
    + generate(): void
    + store(): void
    + download(): void
    + view(): void
}

User --> Question : selects
User --> Topic : selects
User --> PDFDocument : generates
PDFDocument --> Question : contains
PDFDocument --> Topic : contains
@enduml
