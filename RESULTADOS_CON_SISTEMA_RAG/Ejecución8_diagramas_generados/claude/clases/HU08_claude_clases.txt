@startuml
class Student {
    - studentId: String
    - name: String
    - email: String
    + enrollInCourse(): boolean
    + getEnrollments(): List
}

class Course {
    - courseId: String
    - courseName: String
    - description: String
    - capacity: int
    + addStudent(): boolean
    + removeStudent(): boolean
    + getEnrolledStudents(): List
}

class Enrollment {
    - enrollmentId: String
    - enrollmentDate: Date
    - status: String
    + modify(): boolean
    + cancel(): boolean
    + getDetails(): String
}

Student "1" --> "*" Enrollment
Course "1" --> "*" Enrollment
Enrollment --> "1" Student
Enrollment --> "1" Course
@enduml
